```python
# services/factor_analysis.py

class FactorAnalyzer:
    """因子分析器 - 专业版"""
    
    def __init__(self, factor_data, return_data):
        self.factor_data = factor_data
        self.return_data = return_data
    
    def ic_analysis(self, factor_name, period=20):
        """
        IC分析 (信息系数)
        - IC均值
        - IC标准差
        - IR (信息比率)
        - IC胜率
        - IC累积曲线
        """
        ic_series = []
        for date in self.dates:
            factor_vals = self.factor_data.loc[date, factor_name]
            future_ret = self.return_data.loc[date, f'ret_{period}d']
            ic = factor_vals.corr(future_ret, method='spearman')
            ic_series.append(ic)
        
        return {
            'ic_mean': np.mean(ic_series),
            'ic_std': np.std(ic_series),
            'ir': np.mean(ic_series) / np.std(ic_series),
            'ic_win_rate': np.mean([ic > 0 for ic in ic_series]),
            'ic_series': ic_series
        }
    
    def quantile_return(self, factor_name, n_quantiles=10):
        """
        分层收益分析
        - 分10层收益率
        - 多空收益
        - 单调性检验
        """
        returns_by_quantile = {}
        for date in self.dates:
            factor_vals = self.factor_data.loc[date, factor_name]
            quantiles = pd.qcut(factor_vals, n_quantiles, labels=False)
            
            for q in range(n_quantiles):
                stocks_in_q = quantiles[quantiles == q].index
                avg_ret = self.return_data.loc[date, stocks_in_q].mean()
                returns_by_quantile.setdefault(q, []).append(avg_ret)
        
        # 多空收益 (最高层 - 最低层)
        long_short = [returns_by_quantile[n_quantiles-1][i] - 
                     returns_by_quantile[0][i] for i in range(len(self.dates))]
        
        return {
            'quantile_returns': {q: np.mean(rets) for q, rets in returns_by_quantile.items()},
            'long_short_return': np.mean(long_short),
            'long_short_sharpe': np.mean(long_short) / np.std(long_short),
            'monotonicity': self._calc_monotonicity(returns_by_quantile)
        }
    
    def turnover_analysis(self, factor_name, n_quantiles=10):
        """
        换手率分析
        - 各层换手率
        - 换手率稳定性
        """
        turnovers = []
        for i in range(1, len(self.dates)):
            prev_date = self.dates[i-1]
            curr_date = self.dates[i]
            
            factor_prev = self.factor_data.loc[prev_date, factor_name]
            factor_curr = self.factor_data.loc[curr_date, factor_name]
            
            # 计算前后两期的股票变化
            q_prev = pd.qcut(factor_prev, n_quantiles, labels=False)
            q_curr = pd.qcut(factor_curr, n_quantiles, labels=False)
            
            turnover = sum(q_curr != q_prev) / len(q_curr)
            turnovers.append(turnover)
        
        return {
            'turnover_mean': np.mean(turnovers),
            'turnover_std': np.std(turnovers),
            'turnover_series': turnovers
        }
```

```python
# services/barra_model.py

class BarraRiskModel:
    """
    Barra CNE5/CNE6 风险模型
    用于风险分解和组合风险分析
    """
    
    def __init__(self):
        # 行业因子
        self.industry_factors = self._load_industry_factors()
        # 风格因子
        self.style_factors = {
            'beta': self._calc_beta,
            'momentum': self._calc_momentum,
            'size': self._calc_size,
            'earnings_yield': self._calc_earnings_yield,
            'volatility': self._calc_volatility,
            'growth': self._calc_growth,
            'value': self._calc_value,
            'leverage': self._calc_leverage,
            'liquidity': self._calc_liquidity
        }
    
    def calculate_exposure(self, stock_codes, date):
        """计算股票因子暴露"""
        exposures = {}
        for factor_name, calc_func in self.style_factors.items():
            exposures[factor_name] = calc_func(stock_codes, date)
        return pd.DataFrame(exposures, index=stock_codes)
    
    def calculate_covariance(self, date, lookback=252):
        """计算因子协方差矩阵"""
        # 使用指数加权移动平均
        factor_returns = self._get_factor_returns(date, lookback)
        lambda_ = 0.94  # 衰减因子
        weights = [(1-lambda_) * lambda_**i for i in range(lookback)]
        weights = np.array(weights) / sum(weights)
        
        cov_matrix = np.cov(factor_returns.T, aweights=weights)
        return pd.DataFrame(cov_matrix, 
                          index=factor_returns.columns,
                          columns=factor_returns.columns)
    
    def decompose_risk(self, portfolio_weights, date):
        """风险分解"""
        exposures = self.calculate_exposure(portfolio_weights.index, date)
        cov_matrix = self.calculate_covariance(date)
        
        # 组合因子暴露
        portfolio_exposure = (exposures.T * portfolio_weights).T.sum()
        
        # 总风险
        total_var = portfolio_exposure @ cov_matrix @ portfolio_exposure
        
        # 因子风险贡献
        factor_contrib = (portfolio_exposure * (cov_matrix @ portfolio_exposure)) / total_var
        
        return {
            'total_risk': np.sqrt(total_var),
            'factor_exposure': portfolio_exposure,
            'factor_contribution': factor_contrib,
            'diversification_ratio': self._calc_diversification(portfolio_exposure, cov_matrix)
        }
```

```python
# services/technical_indicators.py

class TechnicalIndicators:
    """技术指标库"""
    
    # 趋势指标
    @staticmethod
    def ma(series, window):
        return series.rolling(window=window).mean()
    
    @staticmethod
    def ema(series, window):
        return series.ewm(span=window, adjust=False).mean()
    
    @staticmethod
    def macd(close, fast=12, slow=26, signal=9):
        ema_fast = close.ewm(span=fast, adjust=False).mean()
        ema_slow = close.ewm(span=slow, adjust=False).mean()
        dif = ema_fast - ema_slow
        dea = dif.ewm(span=signal, adjust=False).mean()
        hist = dif - dea
        return dif, dea, hist
    
    @staticmethod
    def bollinger_bands(close, window=20, std_dev=2):
        ma = close.rolling(window=window).mean()
        std = close.rolling(window=window).std()
        upper = ma + std_dev * std
        lower = ma - std_dev * std
        percent_b = (close - lower) / (upper - lower)
        bandwidth = (upper - lower) / ma
        return upper, ma, lower, percent_b, bandwidth
    
    @staticmethod
    def ichimoku_cloud(high, low, close):
        """一目均衡表"""
        tenkan_sen = (high.rolling(9).max() + low.rolling(9).min()) / 2
        kijun_sen = (high.rolling(26).max() + low.rolling(26).min()) / 2
        senkou_span_a = ((tenkan_sen + kijun_sen) / 2).shift(26)
        senkou_span_b = ((high.rolling(52).max() + low.rolling(52).min()) / 2).shift(26)
        chikou_span = close.shift(-26)
        return tenkan_sen, kijun_sen, senkou_span_a, senkou_span_b, chikou_span
    
    # 动量指标
    @staticmethod
    def rsi(close, window=14):
        delta = close.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    @staticmethod
    def kdj(high, low, close, n=9, m1=3, m2=3):
        rsv = (close - low.rolling(n).min()) / (high.rolling(n).max() - low.rolling(n).min()) * 100
        k = rsv.ewm(com=m1-1, adjust=False).mean()
        d = k.ewm(com=m2-1, adjust=False).mean()
        j = 3 * k - 2 * d
        return k, d, j
    
    @staticmethod
    def williams_r(high, low, close, window=14):
        highest_high = high.rolling(window=window).max()
        lowest_low = low.rolling(window=window).min()
        return -100 * (highest_high - close) / (highest_high - lowest_low)
    
    # 波动率指标
    @staticmethod
    def atr(high, low, close, window=14):
        tr1 = high - low
        tr2 = abs(high - close.shift())
        tr3 = abs(low - close.shift())
        tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
        return tr.rolling(window=window).mean()
    
    @staticmethod
    def obv(close, volume):
        """能量潮指标"""
        obv = [0]
        for i in range(1, len(close)):
            if close.iloc[i] > close.iloc[i-1]:
                obv.append(obv[-1] + volume.iloc[i])
            elif close.iloc[i] < close.iloc[i-1]:
                obv.append(obv[-1] - volume.iloc[i])
            else:
                obv.append(obv[-1])
        return pd.Series(obv, index=close.index)
```

```python
# services/quantitative_signals.py

class QuantitativeSignals:
    """量化交易信号生成"""
    
    def __init__(self, data):
        self.data = data
        self.indicators = TechnicalIndicators()
    
    def generate_ma_crossover_signal(self, short=5, long=20):
        """均线金叉死叉信号"""
        ma_short = self.indicators.ma(self.data['close'], short)
        ma_long = self.indicators.ma(self.data['close'], long)
        
        golden_cross = (ma_short > ma_long) & (ma_short.shift(1) <= ma_long.shift(1))
        death_cross = (ma_short < ma_long) & (ma_short.shift(1) >= ma_long.shift(1))
        
        return pd.DataFrame({
            'signal': np.where(golden_cross, 1, np.where(death_cross, -1, 0)),
            'ma_short': ma_short,
            'ma_long': ma_long
        })
    
    def generate_macd_signal(self):
        """MACD信号"""
        dif, dea, hist = self.indicators.macd(self.data['close'])
        
        golden_cross = (dif > dea) & (dif.shift(1) <= dea.shift(1))
        death_cross = (dif < dea) & (dif.shift(1) >= dea.shift(1))
        
        return pd.DataFrame({
            'signal': np.where(golden_cross, 1, np.where(death_cross, -1, 0)),
            'dif': dif,
            'dea': dea,
            'histogram': hist
        })
    
    def generate_rsi_signal(self, overbought=70, oversold=30):
        """RSI超买超卖信号"""
        rsi = self.indicators.rsi(self.data['close'])
        
        overbought_signal = rsi > overbought
        oversold_signal = rsi < oversold
        
        return pd.DataFrame({
            'signal': np.where(oversold_signal, 1, 
                     np.where(overbought_signal, -1, 0)),
            'rsi': rsi,
            'overbought': overbought,
            'oversold': oversold
        })
    
    def generate_bollinger_signal(self):
        """布林带信号"""
        upper, mid, lower, percent_b, bandwidth = self.indicators.bollinger_bands(
            self.data['close']
        )
        
        touch_upper = self.data['close'] >= upper
        touch_lower = self.data['close'] <= lower
        
        return pd.DataFrame({
            'signal': np.where(touch_lower, 1, 
                     np.where(touch_upper, -1, 0)),
            'upper': upper,
            'mid': mid,
            'lower': lower,
            'percent_b': percent_b,
            'bandwidth': bandwidth
        })
    
    def combine_signals(self, signals_dict, weights=None):
        """
        多信号合成
        """
        if weights is None:
            weights = {name: 1.0 for name in signals_dict}
        
        combined = pd.DataFrame(index=self.data.index)
        combined['score'] = 0
        
        for name, signal_df in signals_dict.items():
            combined['score'] += signal_df['signal'] * weights.get(name, 1.0)
        
        # 归一化到 -1 到 1
        max_score = combined['score'].abs().max()
        if max_score > 0:
            combined['score'] = combined['score'] / max_score
        
        # 生成最终信号
        combined['signal'] = np.where(combined['score'] > 0.5, 1,
                            np.where(combined['score'] < -0.5, -1, 0))
        
        return combined
```

```python
# services/portfolio_optimization.py

import cvxpy as cp
import numpy as np
from scipy.optimize import minimize

class PortfolioOptimizer:
    """组合优化器 - 专业版"""
    
    def __init__(self, expected_returns, cov_matrix, risk_free_rate=0.03):
        self.mu = expected_returns
        self.sigma = cov_matrix
        self.rf = risk_free_rate
        self.n_assets = len(expected_returns)
    
    def mean_variance_optimization(self, target_return=None, target_risk=None, allow_short=False):
        """
        马科维茨均值-方差优化
        """
        w = cp.Variable(self.n_assets)
        
        portfolio_return = self.mu.values @ w
        portfolio_variance = cp.quad_form(w, self.sigma.values)
        
        constraints = [cp.sum(w) == 1]
        if not allow_short:
            constraints.append(w >= 0)
        
        if target_return is not None:
            # 最小风险，给定收益
            constraints.append(portfolio_return >= target_return)
            objective = cp.Minimize(portfolio_variance)
        elif target_risk is not None:
            # 最大收益，给定风险
            constraints.append(cp.quad_form(w, self.sigma.values) <= target_risk**2)
            objective = cp.Maximize(portfolio_return)
        else:
            # 最大夏普比率
            objective = cp.Maximize((portfolio_return - self.rf) / cp.sqrt(portfolio_variance))
        
        prob = cp.Problem(objective, constraints)
        prob.solve()
        
        return {
            'weights': w.value,
            'expected_return': portfolio_return.value,
            'risk': np.sqrt(portfolio_variance.value),
            'sharpe_ratio': (portfolio_return.value - self.rf) / np.sqrt(portfolio_variance.value)
        }
    
    def black_litterman(self, views, view_confidences, tau=0.05):
        """
        Black-Litterman 模型
        融合市场均衡收益和投资者观点
        
        Parameters:
        -----------
        views: dict
            投资者观点，如 {'AAPL': 0.15, 'GOOGL': 0.10}
        view_confidences: dict
            观点置信度
        tau: float
            缩放因子
        """
        # 市场均衡权重 (市值加权)
        w_eq = self._get_market_cap_weights()
        
        # 市场均衡收益
        pi = self.sigma.values @ w_eq
        
        # 构建观点矩阵 P 和观点收益 Q
        P = np.zeros((len(views), self.n_assets))
        Q = np.zeros(len(views))
        omega = np.zeros((len(views), len(views)))
        
        for i, (asset, view_return) in enumerate(views.items()):
            asset_idx = list(self.mu.index).index(asset)
            P[i, asset_idx] = 1
            Q[i] = view_return
            omega[i, i] = view_confidences.get(asset, 0.1)
        
        # 后验收益
        sigma_inv = np.linalg.inv(self.sigma.values)
        omega_inv = np.linalg.inv(omega)
        
        M_inv = np.linalg.inv(np.linalg.inv(tau * self.sigma.values) + P.T @ omega_inv @ P)
        
        posterior_return = M_inv @ (np.linalg.inv(tau * self.sigma.values) @ pi + P.T @ omega_inv @ Q)
        posterior_cov = self.sigma.values + M_inv
        
        # 用后验收益重新优化
        self.mu = pd.Series(posterior_return, index=self.mu.index)
        self.sigma = pd.DataFrame(posterior_cov, index=self.sigma.index, columns=self.sigma.columns)
        
        return self.mean_variance_optimization()
    
    def risk_parity(self, target_risk=None):
        """
        风险平价组合
        使各资产对组合风险的贡献相等
        """
        def risk_budget_objective(w):
            portfolio_vol = np.sqrt(w @ self.sigma.values @ w)
            marginal_contrib = (self.sigma.values @ w) / portfolio_vol
            risk_contrib = w * marginal_contrib
            target = portfolio_vol / self.n_assets
            return np.sum((risk_contrib - target)**2)
        
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        bounds = [(0, 1) for _ in range(self.n_assets)]
        
        w0 = np.array([1/self.n_assets] * self.n_assets)
        result = minimize(risk_budget_objective, w0, method='SLSQP', 
                         bounds=bounds, constraints=constraints)
        
        w = result.x
        portfolio_vol = np.sqrt(w @ self.sigma.values @ w)
        
        return {
            'weights': w,
            'expected_return': w @ self.mu.values,
            'risk': portfolio_vol,
            'risk_contributions': self._calc_risk_contributions(w, portfolio_vol)
        }
    
    def maximum_diversification(self):
        """
        最大分散化组合
        最大化分散比率
        """
        stds = np.sqrt(np.diag(self.sigma.values))
        
        def neg_diversification_ratio(w):
            portfolio_vol = np.sqrt(w @ self.sigma.values @ w)
            weighted_vols = w @ stds
            return -weighted_vols / portfolio_vol
        
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        bounds = [(0, 1) for _ in range(self.n_assets)]
        
        w0 = np.array([1/self.n_assets] * self.n_assets)
        result = minimize(neg_diversification_ratio, w0, method='SLSQP',
                         bounds=bounds, constraints=constraints)
        
        w = result.x
        portfolio_vol = np.sqrt(w @ self.sigma.values @ w)
        
        return {
            'weights': w,
            'expected_return': w @ self.mu.values,
            'risk': portfolio_vol,
            'diversification_ratio': (w @ stds) / portfolio_vol
        }
    
    def target_date_glide_path(self, target_date, current_age=None):
        """
        目标日期策略 (生命周期策略)
        随着年龄增长自动调整股债比例
        """
        if current_age:
            years_to_target = 65 - current_age
        else:
            years_to_target = (pd.to_datetime(target_date) - pd.Timestamp.now()).days / 365
        
        # 股票比例随时间递减
        stock_ratio = max(0.2, min(0.9, 0.9 - (65 - years_to_target) * 0.02))
        bond_ratio = 1 - stock_ratio
        
        # 基于比例约束优化
        w = cp.Variable(self.n_assets)
        
        # 假设前一半是股票，后一半是债券
        stock_assets = self.n_assets // 2
        
        portfolio_variance = cp.quad_form(w, self.sigma.values)
        
        constraints = [
            cp.sum(w) == 1,
            w >= 0,
            cp.sum(w[:stock_assets]) >= stock_ratio - 0.05,
            cp.sum(w[:stock_assets]) <= stock_ratio + 0.05
        ]
        
        prob = cp.Problem(cp.Minimize(portfolio_variance), constraints)
        prob.solve()
        
        return {
            'weights': w.value,
            'stock_ratio': stock_ratio,
            'bond_ratio': bond_ratio,
            'expected_return': self.mu.values @ w.value,
            'risk': np.sqrt(portfolio_variance.value)
        }
    
    def cppi_strategy(self, floor=0.8, multiplier=3):
        """
        CPPI (固定比例投资组合保险)
        保本策略
        """
        total_value = 1.0
        floor_value = floor * total_value
        cushion = total_value - floor_value
        
        # 风险资产敞口 = 乘数 × 缓冲
        risky_exposure = min(multiplier * cushion, total_value)
        safe_exposure = total_value - risky_exposure
        
        # 分配到具体资产
        # 假设第一个资产是风险资产，其余是无风险资产
        w = np.zeros(self.n_assets)
        w[0] = risky_exposure
        w[1:] = safe_exposure / (self.n_assets - 1)
        
        return {
            'weights': w,
            'risky_exposure': risky_exposure,
            'safe_exposure': safe_exposure,
            'floor': floor_value,
            'cushion': cushion
        }
    
    def efficient_frontier(self, n_points=50):
        """
        计算有效前沿
        """
        min_return = self.mu.min()
        max_return = self.mu.max()
        target_returns = np.linspace(min_return, max_return, n_points)
        
        frontier = []
        for target in target_returns:
            try:
                result = self.mean_variance_optimization(target_return=target)
                frontier.append({
                    'return': result['expected_return'],
                    'risk': result['risk'],
                    'sharpe': result['sharpe_ratio'],
                    'weights': result['weights']
                })
            except:
                continue
        
        return frontier
```

```python
# services/risk_management.py

class RiskManager:
    """风险管理 - 专业版"""
    
    def __init__(self, returns, weights=None):
        self.returns = returns
        self.weights = weights
        self.portfolio_returns = self._calc_portfolio_returns()
    
    def calculate_var(self, confidence=0.95, method='historical'):
        """
        计算VaR (风险价值)
        
        Parameters:
        -----------
        confidence: float
            置信水平，默认95%
        method: str
            'historical' - 历史模拟法
            'parametric' - 参数法 (正态分布)
            'monte_carlo' - 蒙特卡洛模拟
        """
        if method == 'historical':
            return np.percentile(self.portfolio_returns, (1-confidence)*100)
        
        elif method == 'parametric':
            mean = np.mean(self.portfolio_returns)
            std = np.std(self.portfolio_returns)
            z_score = stats.norm.ppf(1-confidence)
            return mean + z_score * std
        
        elif method == 'monte_carlo':
            # 蒙特卡洛模拟
            simulated_returns = np.random.normal(
                np.mean(self.portfolio_returns),
                np.std(self.portfolio_returns),
                100000
            )
            return np.percentile(simulated_returns, (1-confidence)*100)
    
    def calculate_cvar(self, confidence=0.95):
        """
        计算CVaR/ES (条件风险价值/期望损失)
        超过VaR的平均损失
        """
        var = self.calculate_var(confidence)
        return np.mean(self.portfolio_returns[self.portfolio_returns <= var])
    
    def calculate_max_drawdown(self):
        """
        计算最大回撤
        """
        cumulative = (1 + self.portfolio_returns).cumprod()
        running_max = cumulative.expanding().max()
        drawdown = (cumulative - running_max) / running_max
        
        return {
            'max_drawdown': drawdown.min(),
            'drawdown_series': drawdown,
            'drawdown_start': self._find_drawdown_periods(drawdown)
        }
    
    def calculate_stress_test(self, scenarios):
        """
        压力测试
        
        Parameters:
        -----------
        scenarios: dict
            压力情景，如：
            {
                'market_crash': {'return': -0.30, 'correlation': 0.9},
                'interest_rate_shock': {'return': -0.15, 'correlation': 0.7},
                'liquidity_crisis': {'return': -0.20, 'correlation': 0.8}
            }
        """
        results = {}
        
        for scenario_name, params in scenarios.items():
            # 调整收益和相关性
            adjusted_returns = self.returns * (1 + params['return'])
            
            # 重新计算组合收益
            portfolio_return = np.dot(self.weights, adjusted_returns.mean())
            
            results[scenario_name] = {
                'portfolio_return': portfolio_return,
                'var_95': self.calculate_var(confidence=0.95),
                'max_drawdown': self.calculate_max_drawdown()['max_drawdown']
            }
        
        return results
    
    def calculate_tracking_error(self, benchmark_returns):
        """
        计算跟踪误差 (对被动管理)
        """
        active_returns = self.portfolio_returns - benchmark_returns
        return {
            'tracking_error': np.std(active_returns) * np.sqrt(252),
            'information_ratio': np.mean(active_returns) / np.std(active_returns),
            'up_capture': self._calc_capture_ratio(benchmark_returns, 'up'),
            'down_capture': self._calc_capture_ratio(benchmark_returns, 'down')
        }
```

```python
        if quantity <= 0:

        else:
            executed_price = price * (1 - self.slippage)
        
        # 考虑冲击成本 (大额订单)
        trade_value = quantity * executed_price
        if trade_value > 1000000:  # 大于100万

                executed_price *= (1 + impact)
            else:
                executed_price *= (1 - impact)

        if not is_open:
            prev_close = self._get_prev_close(stock_code)
            if direction == 'BUY' and executed_price >= prev_close * 1.1:
                return  # 涨停无法买入
            if direction == 'SELL' and executed_price <= prev_close * 0.9:
                return  # 跌停无法卖出
        
        # 考虑成交量限制
        max_trade_volume = int(volume * 0.1)  # 最多成交当日成交量的10%
        quantity = min(quantity, max_trade_volume)
```

```python
# strategies/multi_factor_strategy.py

class MultiFactorStrategy(BaseStrategy):
    """
    多因子选股策略
    """
    
    def __init__(self, factors, weights=None, n_stocks=20, holding_period=20):
        super().__init__()
        self.factors = factors
        self.weights = weights or {f: 1.0 for f in factors}
        self.n_stocks = n_stocks
        self.holding_period = holding_period
        self.last_rebalance = None
    
    def generate_signals(self, data, current_positions):
        """
        基于多因子打分选股
        """
        # 计算因子得分
        scores = pd.DataFrame(index=data.index)
        
        for factor in self.factors:
            if factor in data.columns:
                # 因子方向：1为正向，-1为负向
                direction = self.weights.get(factor, 1)
                factor_data = data[factor]
                
                # 缺失值处理
                factor_data = factor_data.fillna(factor_data.median())
                
                # 标准化
                factor_data = (factor_data - factor_data.mean()) / factor_data.std()
                
                scores[factor] = factor_data * direction
        
        # 合成总分
        scores['total'] = scores.sum(axis=1)
        
        # 选出前N只股票
        top_stocks = scores.nlargest(self.n_stocks, 'total').index.tolist()
        
        # 生成信号
        signals = {}
        for code in data.index:
            if code in top_stocks:
                signals[code] = {'signal': 1, 'weight': 1.0 / self.n_stocks}
            elif code in current_positions:
                signals[code] = {'signal': -1, 'weight': 0}
            else:
                signals[code] = {'signal': 0, 'weight': 0}
        
        return signals

class TechnicalStrategy(BaseStrategy):
    """
    技术分析策略
    支持多指标组合
    """
    
    def __init__(self, indicators=None, combine_method='majority'):
        super().__init__()
        self.indicators = indicators or ['ma_cross', 'macd']
        self.combine_method = combine_method
    
    def generate_signals(self, data, current_positions):
        signals_dict = {}
        
        for indicator in self.indicators:
            if indicator == 'ma_cross':
                signals_dict[indicator] = self._ma_cross_signal(data)
            elif indicator == 'macd':
                signals_dict[indicator] = self._macd_signal(data)
            elif indicator == 'rsi':
                signals_dict[indicator] = self._rsi_signal(data)
            # ... 更多指标
        
        # 合成信号
        final_signals = self._combine_signals(signals_dict)
        
        return final_signals
    
    def _combine_signals(self, signals_dict):
        """多信号合成"""
        if self.combine_method == 'majority':
            # 多数投票
            combined = pd.DataFrame(signals_dict).mode(axis=1)
        elif self.combine_method == 'unanimous':
            # 全体一致
            combined = pd.DataFrame(signals_dict).apply(
                lambda x: x.iloc[0] if all(x == x.iloc[0]) else 0, axis=1
            )
        else:
            # 加权平均
            weights = [1.0] * len(signals_dict)
            combined = pd.DataFrame(signals_dict).dot(weights) / sum(weights)
            combined = np.where(combined > 0.5, 1, 
                       np.where(combined < -0.5, -1, 0))
        
        return combined.to_dict()
```
